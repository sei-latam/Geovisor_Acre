# Definición de variables globales de entorno para la automatización de la API
USER="admin"
PASS="geoserver"
WS="rio_acre_manchas"
BASE_DIR="/usr/share/geoserver/manchas_inundacion"

# Recorrer cada directorio de periodo de retorno
for dir in tr02 tr10 tr50 tr100 tr2023; do
    # Continuar con la siguiente iteración si el directorio físico no existe
    [ -d "$BASE_DIR/$dir" ] || continue
    cd "$BASE_DIR/$dir"
    
    # Buscar todos los archivos con extensión .tif dentro de la carpeta
    for file in *.tif; do
        [ -e "$file" ] || break
        layer_name="${file%.tif}"
        
        # Enviar petición HTTP POST a la API REST construyendo un almacén tipo GeoTIFF
        curl -s -u "$USER:$PASS" -X POST -H "Content-type: application/xml" -d "<coverageStore><name>$layer_name</name><type>GeoTIFF</type><enabled>true</enabled><workspace>$WS</workspace><url>file:$BASE_DIR/$dir/$file</url></coverageStore>" "http://localhost:8080/geoserver/rest/workspaces/$WS/coveragestores"
        
        # Respiro para que GeoServer guarde en disco sin saturarse
        sleep 0.3
    done
done


# Bucle de publicación masiva de coberturas cartográficas
for dir in tr02 tr10 tr50 tr100 tr2023; do
    [ -d "$BASE_DIR/$dir" ] || continue
    cd "$BASE_DIR/$dir"
    for file in *.tif; do
        [ -e "$file" ] || break
        layer_name="${file%.tif}"
        
        # Enviar petición POST para activar la capa estableciendo las proyecciones y políticas
        curl -s -u "$USER:$PASS" -X POST -H "Content-type: application/xml" -d "<coverage><name>$layer_name</name><nativeName>$layer_name</nativeName><title>$layer_name</title><nativeCRS>EPSG:32719</nativeCRS><srs>EPSG:4326</srs><projectionPolicy>REPROJECT_TO_DECLARED</projectionPolicy><enabled>true</enabled></coverage>" "http://localhost:8080/geoserver/rest/workspaces/$WS/coveragestores/$layer_name/coverages"
        
        # Pausa de seguridad para sincronización del catálogo
        sleep 0.3
    done
done

<StyledLayerDescriptor version="1.0.0" xmlns="http://www.opengis.net/sld" xmlns:ogc="http://www.opengis.net/ogc" xmlns:xlink="http://www.w3.org/1999/xlink" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:schemaLocation="http://www.opengis.net/sld StyledLayerDescriptor.xsd">
  <NamedLayer>
   <Name>estilo_inundacion</Name>
   <UserStyle>
     <Title>Rampa de Inundacion (Depth)</Title>
     <FeatureTypeStyle>
       <Rule>
          <RasterSymbolizer>
            <ColorMap type="ramp">
              <ColorMapEntry color="#ffffff" quantity="0" label="0 metros" opacity="0.0"/>
              <ColorMapEntry color="#a6bddb" quantity="2" label="2 metros" opacity="0.6"/>

       </Rule>

   </UserStyle>

EOF

# 2. Enviar el archivo SLD generado como cuerpo de la petición (usando @) para escribir la definición de la rampa de color
